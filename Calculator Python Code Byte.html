def exponent():
    """A function dedicated to exponenting a number x I.E X^Y-power"""
    try:
        num_one = float(input("Enter your number: "))
        power = float("Enter the power: ")
        # Returns a number to the power of the power var
        return num_one ** power
    except:
        return "Invalid"


def root_num():
    """A function dedicated to rooting a number x I.E X^1/z-power"""
    try:
        num_one = float(input('Enter your number: '))
        root = 1 / (float(input('Enter your root: ')))
        return num_one ** root
    except:
        return "Invalid"


def degree_radian_conversion():
    """A fuction dedicated to the coversion of degrees(x) to radians I.E x *π/180 , and Radians(x) to degrees I.E x*180/π """
    degree_or_radian = input(
        "1.Degree to Radian\n2. Radian to Degree\nChoose: ")
    if degree_or_radian == '1':
        degree = float(input('Enter Degree: '))
        return degree * (math.pi / 180)
    elif degree_or_radian == '2':
        radian = float(input('Enter Radian: '))
        # Returns a value after being pluggeed into the Radian -> Degree conversion formula
        return radian * (180/math.pi)
    else:
        # While loop handling Invalid Input
        while degree_or_radian != "1" or degree_or_radian != "2":
            print('Invalid Input')
            degree_or_radian = input(
                "1.Degree to Radian\n2. Radian to Degree\nChoose: ")
            if degree_or_radian == '1':
                degree = float(input('Enter Degree: '))
                return degree * (math.pi / 180)
            elif degree_or_radian == '2':
                radian = float(input('Enter Radian: '))
                return radian * (180/math.pi)

def graph():
    """A fuction that takes an equation and graphs it"""

    # Sets x values
    x = np.arange(-1000, 1000, 0.01)
    # 1-4 Relating to Conditionals for Graph types
    graph_type = input(
        '1. Linear\n2. Quadratic\n3. Cubic\n4. Quartic\nChoose: ')
    # Y/N on x-axis reflection
    negative = input('Negative? y/n ')
    if negative == 'y':
        negative = -1
    elif negative == 'n':
        negative = 1
    # Invalid Input for Negative var
    else:
        while negative != "y" or negative != "n":
            print('Invalid Input')
            negative = input('y/n')
            if negative == 'y':
                negative = -1
            elif negative == 'n':
                negative = 1
    # Hor/Ver Shrink/Stretch input var used in equations
    strech_shrink = float(
        input("By what int value would you like to stretch or shrink your graph: "))
    # Conditionals Set for Graph Type:
    # Linear
    if graph_type == "1":
        y_int = float(input("Y-int: "))

    elif graph_type == "2":

        hor_trans = float(input('Enter your horizontal transformation: '))

        plt.show()
    # Cubic
    elif graph_type == "3":

    # Quartic
